import { RadioButton } from './radio-button';
import { Figma } from '@koodoo/koomeleon.figma';
import { useState } from 'react';

### Component usage

```js
const [selected, setSelected] = useState(null);

const radioInput = {
  primaryText: 'First Option',
  secondaryText: 'Extra text providing additional context',
  value: 'Demo Option',
  onValueChange: (value) => setSelected(value),
  suffix: {
    value: 'Value',
    label: 'Label',
  },
  error: false,
};

return (
      <RadioButton {...radioInput} />
);
```

### Using props to customize the text

Modify the text to see it change live:

```js live
() => {
  const [selected, setSelected] = useState(null);

  const radioInput = {
    primaryText: 'First Option',
    secondaryText: 'Extra text providing additional context',
    value: 'Demo Option',
    onValueChange: (value) => setSelected(value),
    suffix: {
      valueText: 'Value',
      valueLabelText: 'Label',
    },
    error: false,
  };

  return (
    <>
        <RadioButton {...radioInput} />
      <div>Selected: {selected ? selected : 'None'}</div>
    </>
  );
};
```
