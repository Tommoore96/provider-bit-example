import { RadioGroup } from './radio-group';
import { useState, useEffect } from 'react';
import { Figma } from '@koodoo/koomeleon.figma';

### Component usage

```js
const [selectedValue, setSelectedValue] = useState('None');
const radioButtons = [
  {
    primaryText: 'First Option',
    error: false,
    value: 'First Option',
  },
  {
    primaryText: 'Second Option',
    secondaryText: 'Extra text providing additional context',
    error: false,
    value: 'Second Option',
  },
  {
    primaryText: 'Third Option',
    secondaryText: 'Extra text providing additional context',
    suffix: {
      valueText: 'Value',
      valueLabelText: 'Label',
    },
    error: false,
    value: 'Third Option',
  },
];

const props = {
  fieldSet: 'LiveExample',
  radioButtons: radioButtons,
  errorMessage: '',
  onValueChange: (value) => setSelectedValue(value),
};

return (
      <RadioGroup {...props} />
);
```
