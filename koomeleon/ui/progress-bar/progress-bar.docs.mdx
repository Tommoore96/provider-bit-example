labels: ['Progress Bar']

import { ProgressBar } from './progress-bar';
import { useEffect, useState } from 'react';
import { Button } from '@koodoo/koomeleon.ui.button';
import { Figma } from '@koodoo/koomeleon.figma';

### Component usage

```js
() => {
  const [percent, setPercent] = useState(50);
  return (
    <ProgressBar percent={percent} showPercent={false}/>
    <button  onClick={() => percent >= 100 ? setPercent(0) : setPercent(100)}>Animate</button>
  )
};
```

### Using props to customize the text

Modify the text to see it change live:

```js live
() => {
  const [percent, setPercent] = useState(50);
  return (
      <div style={{ width: '350px' }}>
        <ProgressBar percent={percent} showPercent={false} />
        <button
          onClick={() => (percent >= 100 ? setPercent(0) : setPercent(100))}
        >
          Animate
        </button>
      </div>
  );
};
```
